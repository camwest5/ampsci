171 inline double cg_2(int two_j1, int two_m1, int two_j2, int two_m2, int two_J,
172  int two_M)
173 // <j1 m1, j2 m2 | J M> = (-1)^(j1-j2+M) * std::sqrt(2J+1) * (j1 j2 J)
174 // . (m1 m2 -M)
175 // (Last term is 3j symbol)
176 // Note: this function takes INTEGER values, that have already multiplied by 2!
177 // Works for l and j (integer and half-integer)
178 {
179  if (triangle(two_j1, two_j2, two_J) * sumsToZero(two_m1, two_m2, -two_M) == 0)
180  return 0;
181  int sign = -1;
182  if ((two_j1 - two_j2 + two_M) % 4 == 0)
183  sign = 1; // mod 4 (instead 2), since x2
184  return sign * std::sqrt(two_J + 1.) *
185  gsl_sf_coupling_3j(two_j1, two_j2, two_J, two_m1, two_m2, -two_M);
186 }

267 inline double Ck_kk(int k, int ka, int kb)
268 // Reduced (relativistic) angular ME:
269 // <ka||C^k||kb> = (-1)^(ja+1/2) * srt([ja][jb]) * 3js(ja jb k, -1/2 1/2 0) * Pi
270 // Note: takes in kappa! (not j!)
271 {
272  if (parity(l_k(ka), l_k(kb), k) == 0) {
273  return 0.0;
274  }
275  auto two_ja = twoj_k(ka);
276  auto two_jb = twoj_k(kb);
277  // auto sign = ((two_ja + 1) / 2 % 2 == 0) ? 1 : -1;
278  auto sign = evenQ_2(two_ja + 1) ? 1 : -1;
279  auto f = std::sqrt((two_ja + 1) * (two_jb + 1));
280  auto g = special_threej_2(two_ja, two_jb, 2 * k);
281  return sign * f * g;
282 }

311  // j = |k|-0.5
312  // kmin = |ja-jb| = | |ka| - |kb| |

340 inline double S_kk(int ka, int kb) {
341  auto la = l_k(ka);
342  if (la != l_k(kb))
343  return 0.0;
344  auto tja = twoj_k(ka);
345  auto tjb = twoj_k(kb);
346  if (triangle(tja, 2, tjb) == 0)
347  return 0;
348  auto sign = (((tja + 2 * la + 3) / 2) % 2 == 0) ? 1 : -1;
349  auto f = std::sqrt((tja + 1) * (tjb + 1) * 1.5);
350  // auto sixj = gsl_sf_coupling_6j(tja, 2, tjb, 1, 2 * la, 1);
351  auto sixj_sign = ((ka + kb) % 2 == 0) ? 1.0 : -1.0;
352  auto sixj = 0.5 * sixj_sign *
353  std::sqrt(std::fabs(((ka - 1) * (ka - 1) - kb * kb) /
354  (3.0 * ka * (1.0 + 2.0 * ka))));
355  return sign * f * sixj;
356 }
