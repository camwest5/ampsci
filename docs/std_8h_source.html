142 template <typename Char, typename OutputIt, typename T>
143 auto write_variant_alternative(OutputIt out, const T &v) -> OutputIt {
144  if constexpr (is_string<T>::value)
145  return write_escaped_string<Char>(out, detail::to_string_view(v));
146  else if constexpr (std::is_same_v<T, Char>)
147  return write_escaped_char(out, v);
148  else
149  return write<Char>(out, v);
150 }

210 template <typename T, typename Char>
211 struct formatter<
212  T, Char,
213  typename std::enable_if<std::is_base_of<std::exception, T>::value>::type> {
214 private:
215  bool with_typename_ = false;
216 
217 public:
218  FMT_CONSTEXPR auto parse(basic_format_parse_context<Char> &ctx)
219  -> decltype(ctx.begin()) {
220  auto it = ctx.begin();
221  auto end = ctx.end();
222  if (it == end || *it == '}')
223  return it;
224  if (*it == 't') {
225  ++it;
226  with_typename_ = true;
227  }
228  return it;
229  }
230 
231  template <typename OutputIt>
232  auto format(const std::exception &ex,
233  basic_format_context<OutputIt, Char> &ctx) const -> OutputIt {
234  format_specs<Char> spec;
235  auto out = ctx.out();
236  if (!with_typename_)
237  return detail::write_bytes(out, string_view(ex.what()), spec);
238 
239  const std::type_info &ti = typeid(ex);
240 #ifdef FMT_HAS_ABI_CXA_DEMANGLE
241  int status = 0;
242  std::size_t size = 0;
243  std::unique_ptr<char, decltype(&std::free)> demangled_name_ptr(
244  abi::__cxa_demangle(ti.name(), nullptr, &size, &status), &std::free);
245 
246  string_view demangled_name_view;
247  if (demangled_name_ptr) {
248  demangled_name_view = demangled_name_ptr.get();
249 
250  // Normalization of stdlib inline namespace names.
251  // libc++ inline namespaces.
252  // std::__1::* -> std::*
253  // std::__1::__fs::* -> std::*
254  // libstdc++ inline namespaces.
255  // std::__cxx11::* -> std::*
256  // std::filesystem::__cxx11::* -> std::filesystem::*
257  if (demangled_name_view.starts_with("std::")) {
258  char *begin = demangled_name_ptr.get();
259  char *to = begin + 5; // std::
260  for (char *from = to, *end = begin + demangled_name_view.size();
261  from < end;) {
262  // This is safe, because demangled_name is NUL-terminated.
263  if (from[0] == '_' && from[1] == '_') {
264  char *next = from + 1;
265  while (next < end && *next != ':')
266  next++;
267  if (next[0] == ':' && next[1] == ':') {
268  from = next + 2;
269  continue;
270  }
271  }
272  *to++ = *from++;
273  }
274  demangled_name_view = {begin, detail::to_unsigned(to - begin)};
275  }
276  } else {
277  demangled_name_view = string_view(ti.name());
278  }
279  out = detail::write_bytes(out, demangled_name_view, spec);
280 #elif FMT_MSC_VERSION
281  string_view demangled_name_view(ti.name());
282  if (demangled_name_view.starts_with("class "))
283  demangled_name_view.remove_prefix(6);
284  else if (demangled_name_view.starts_with("struct "))
285  demangled_name_view.remove_prefix(7);
286  out = detail::write_bytes(out, demangled_name_view, spec);
287 #else
288  out = detail::write_bytes(out, string_view(ti.name()), spec);
289 #endif
290  out = detail::write<Char>(out, Char(':'));
291  out = detail::write<Char>(out, Char(' '));
292  out = detail::write_bytes(out, string_view(ex.what()), spec);
293 
294  return out;
295  }
296 };
