26 template <typename C, typename... Args>
27 inline double integrate(const double dt, std::size_t beg, std::size_t end,
28  const C &f1, const Args &...rest)
29 // // Note: includes no safety checks!
30 // // Integrates from (point) beg to end-1 (i.e., not including end)
31 // // Require:
32 // // * (beg-end) > 2*Nquad
33 // // * end - Nquad > Nquad
34 // // * beg + 2*Nquad
35 // // NB: end-point corrections only applied if beg < Nquad, end > max - nquad
36 // // Not sure if this is best choice - but it ensures that:
37 // // int_{a->b} + int_{b->c} = int_{a->c}
38 {
39 
40  const auto max_grid = f1.size();
41  if (end == 0)
42  end = max_grid;
43  const auto end_mid = std::min(max_grid - Nquad, end);
44  const auto start_mid = std::max(Nquad, beg);
45 
46  double Rint_ends = qip::inner_product_sub(beg, Nquad, cq, f1, rest...);
47 
48  const double Rint_mid =
49  qip::inner_product_sub(start_mid, end_mid, f1, rest...);
50 
51  for (auto i = end_mid; i < end; ++i) {
52  Rint_ends += cq[end_mid + Nquad - i - 1] * qip::multiply_at(i, f1, rest...);
53  }
54  return (Rint_mid + dq_inv * Rint_ends) * dt;
55 }

58 template <typename T>
59 inline std::vector<T> derivative(const std::vector<T> &f,
60  const std::vector<T> &drdt, const T dt,
61  const int order = 1)
62 // df/dr = df/dt * dt/dr = (df/dt) / (dr/dt)
63 // = (df/di) * (di/dt) / (dr/dt)
64 // = (df/di) / (dt * dr/dt)
65 // coeficients from: http://en.wikipedia.org/wiki/Finite_difference_coefficient
66 {
67 
68  std::size_t num_points = f.size();
69  std::vector<T> df(num_points);
70  if (num_points < 4)
71  return df;
72 
73  df[0] = (f[1] - f[0]) / (dt * drdt[0]);
74  df[num_points - 1] =
75  (f[num_points - 1] - f[num_points - 2]) / (dt * drdt[num_points - 1]);
76 
77  df[1] = (f[2] - f[0]) / (2 * dt * drdt[1]);
78  df[num_points - 2] =
79  (f[num_points - 1] - f[num_points - 3]) / (2 * dt * drdt[num_points - 2]);
80 
81  df[2] = (f[0] - 8 * f[1] + 8 * f[3] - f[4]) / (12 * dt * drdt[2]);
82  df[num_points - 3] = (f[num_points - 5] - 8 * f[num_points - 4] +
83  8 * f[num_points - 2] - f[num_points - 1]) /
84  (12 * dt * drdt[num_points - 3]);
85 
86  df[3] = (-1 * f[0] + 9 * f[1] - 45 * f[2] + 45 * f[4] - 9 * f[5] + 1 * f[6]) /
87  (60 * dt * drdt[3]);
88  df[num_points - 4] =
89  (-1 * f[num_points - 7] + 9 * f[num_points - 6] - 45 * f[num_points - 5] +
90  45 * f[num_points - 3] - 9 * f[num_points - 2] + 1 * f[num_points - 1]) /
91  (60 * dt * drdt[num_points - 4]);
92 
93  for (std::size_t i = 4; i < (num_points - 4); i++) {
94  df[i] = ((1.0 / 8) * f[i - 4] - (4.0 / 3) * f[i - 3] + 7 * f[i - 2] -
95  28 * f[i - 1] - (1.0 / 8) * f[i + 4] + (4.0 / 3) * f[i + 3] -
96  7 * f[i + 2] + 28 * f[i + 1]) /
97  (35 * dt * drdt[i]);
98  }
99 
100  if (order > 1)
101  df = derivative(df, drdt, dt, order - 1);
102 
103  return df;
104 }

208 inline double num_integrate(const std::function<double(double)> &f, double a,
209  double b, long unsigned n_pts,
210  t_grid type = linear) {
211  //
212  if (a >= b || n_pts <= 1)
213  return 0.0;
214  const auto dt = (type == linear) ? (b - a) / double(n_pts - 1) :
215  std::log(b / a) / double(n_pts - 1);
216 
217  std::function<double(long unsigned)> x =
218  (type == linear) ? linx(a, dt) : logx(a, dt);
219  std::function<double(long unsigned)> dxdt =
220  (type == linear) ? one() : logx(a, dt);
221 
222  double Rint_s = 0.0;
223  for (long unsigned i = 0; i < Nquad; i++) {
224  Rint_s += cq[i] * f(x(i)) * dxdt(i);
225  }
226 
227  double Rint_m = 0.0;
228  for (auto i = Nquad; i < n_pts - Nquad; i++) {
229  Rint_m += f(x(i)) * dxdt(i);
230  }
231 
232  double Rint_e = 0;
233  for (auto i = n_pts - Nquad; i < n_pts; i++) {
234  Rint_e += cq[n_pts - i - 1] * f(x(i)) * dxdt(i);
235  }
236 
237  return (Rint_m + dq_inv * (Rint_s + Rint_e)) * dt;
238 }
