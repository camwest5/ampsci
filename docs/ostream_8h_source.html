89 inline bool write_ostream_unicode(std::ostream &os, fmt::string_view data) {
90 #if FMT_MSC_VERSION
91  if (auto *buf = dynamic_cast<std::filebuf *>(os.rdbuf()))
92  if (FILE *f = get_file(*buf))
93  return write_console(f, data);
94 #elif defined(_WIN32) && defined(__GLIBCXX__)
95  auto *rdbuf = os.rdbuf();
96  FILE *c_file;
97  if (auto *sfbuf = dynamic_cast<__gnu_cxx::stdio_sync_filebuf<char> *>(rdbuf))
98  c_file = sfbuf->file();
99  else if (auto *fbuf = dynamic_cast<__gnu_cxx::stdio_filebuf<char> *>(rdbuf))
100  c_file = fbuf->file();
101  else
102  return false;
103  if (c_file)
104  return write_console(c_file, data);
105 #elif defined(_WIN32) && defined(_LIBCPP_VERSION)
106  if (auto *buf = dynamic_cast<std::__stdoutbuf<char> *>(os.rdbuf()))
107  if (FILE *f = get_file(*buf))
108  return write_console(f, data);
109 #else
110  ignore_unused(os, data);
111 #endif
112  return false;
113 }

119 // Write the content of buf to os.
120 // It is a separate function rather than a part of vprint to simplify testing.
121 template <typename Char>
122 void write_buffer(std::basic_ostream<Char> &os, buffer<Char> &buf) {
123  const Char *buf_data = buf.data();
124  using unsigned_streamsize = std::make_unsigned<std::streamsize>::type;
125  unsigned_streamsize size = buf.size();
126  unsigned_streamsize max_size = to_unsigned(max_value<std::streamsize>());
127  do {
128  unsigned_streamsize n = size <= max_size ? size : max_size;
129  os.write(buf_data, static_cast<std::streamsize>(n));
130  buf_data += n;
131  size -= n;
132  } while (size != 0);
133 }
