232 std::vector<T> uniform_range(T first, T last, N number) {
233  static_assert(std::is_arithmetic_v<T>,
234  "In uniform_range(T, T, N), T must be arithmetic");

236  "In uniform_range(T, T, N), N must be integral");
237  assert(number >= 2);
238 
239  std::vector<T> range;
240  range.reserve(static_cast<std::size_t>(number));
241  const auto interval = static_cast<double>(last - first);
242  range.push_back(first); // guarentee first is first
243  for (N i = 1; i < number - 1; ++i) {
244  const auto eps = static_cast<double>(i) / static_cast<double>(number - 1);
245  const auto value = static_cast<double>(first) + (eps * interval);
246  range.push_back(static_cast<T>(value));
247  }
248  range.push_back(last); // guarentee last is last

285 template <typename T, typename N>
286 std::vector<T> loglinear_range(T first, T last, T b, N number) {
287  static_assert(std::is_floating_point_v<T>,
288  "In loglinear_range(T, T, T, N), T must be floating point");
289  static_assert(std::is_integral_v<N>,
290  "In loglinear_range(T, T, T, N), N must be integral");
291  assert(number >= 3);
292  assert(first > 0.0 && last > 0.0 && b > 0.0);
293 
294  std::vector<T> range;
295  range.reserve(static_cast<std::size_t>(number));
296 
297  const auto du = (last - first + b * std::log(last / first)) /
298  (static_cast<T>(number - 1));
299 
300  auto next_r = [b](auto u1, auto r_guess) {
301  // Solve eq. u = r + b ln(r) to find r
302  // Use Newtons method
303  // => f(r) = r + b ln(r) - u = 0
304  // dfdr = b(1/r + 1/(b+r))
305  const auto f_u = [b, u1](double tr) { return tr + b * std::log(tr) - u1; };
306  const auto dr = 0.1 * r_guess;
307  const auto delta_targ = r_guess * 1.0e-18;
308  const auto [ri, delta_r] = qip::Newtons(f_u, r_guess, delta_targ, dr, 30);
309  return ri;
310  };
311 
312  auto u = first + b * std::log(first);
313  range.push_back(first);
314  for (N i = 1; i < number - 1; ++i) {
315  u += du;
316  range.push_back(next_r(u, range.back()));
317  }
318  range.push_back(last);
319 
320  return range;
321 }
